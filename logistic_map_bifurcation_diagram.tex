\usepackage{luacode}

\begin{luacode*}
  function logistic()

    local function map(r, x)
      return r*x*(1-x)
    end

    for r=2.5, 4.0, 0.005 do
      local x = 0.1
      for i=1, 100 do
        x = map(r, x)
      end

      for i=1, 150 do
        x = map(r, x)

        tex.sprint(string.format("(%f,%f) ", r, x))
      end

    end

  end
\end{luacode*}
