\documentclass[tikz]{standalone}
\usepackage{amsmath}
\usepackage{tikz}
\usepackage{tikz-3dplot}
\usetikzlibrary{3d}

% Set 3D plot view angle
\tdplotsetmaincoords{75}{75}

\def\rvec{1}

\begin{document}

\begin{tikzpicture}[tdplot_main_coords, scale=2]

  % Shaded ball (not working in svg)
  %\shade[ball color=lightgray, opacity=0.5] (0,0,0) circle (1cm);
  \fill[lightgray!40] (0,0,0) circle (\rvec cm);

  % Great circles
  \tdplotsetrotatedcoords{90}{0}{0};
  \draw[canvas is xy plane at z=0, red, dashed] (0,0) circle (\rvec cm);
  \draw[canvas is zx plane at y=0, green, dashed] (0,0) circle (\rvec cm);
  \draw[blue] (0,0,0) circle (\rvec cm);

  % Points
  \draw plot [mark=*, mark size=0.5] (0, 0, \rvec) node[anchor=south] {\scriptsize $\infty$};
  \draw plot [mark=*, mark size=0.5] (0, 0, -\rvec) node[anchor=north] {\scriptsize $0$};

  \draw plot [mark=*, mark size=0.5] (0, \rvec, 0) node[anchor=west] {\scriptsize $1$};
  \draw plot [mark=*, mark size=0.5] (0, -\rvec, 0) node[anchor=east] {\scriptsize $-1$};

  \draw plot [mark=*, mark size=0.5] (\rvec, 0, 0) node[anchor=north west] {\scriptsize $i$};
  \draw plot [mark=*, mark size=0.5] (-\rvec, 0, 0) node[anchor=south east] {\scriptsize $-i$};

\end{tikzpicture}

\end{document}
